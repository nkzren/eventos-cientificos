-- 2. Selecionar o evento com o menor valor de taxa de patroc√≠nio

select *
from eventos.evento ev
join eventos.patrocina pt on (ev.nome = pt.nome and ev.edicao = pt.edicao)
join eventos.taxa_de_patrocinio tx on (tx.id_taxa = pt.id_taxa)
where tx.valor_minimo <= all (
	select valor_minimo
	from eventos.taxa_de_patrocinio);
